# Sample Linux Commands Executed Inside the Docker Container

# 1. Check current directory
pwd

# 2. List files with details
ls -la

# 3. Create a new directory
mkdir test-folder

# 4. Navigate into the directory
cd test-folder

# 5. Create a sample file
echo "Hello from inside Docker" > hello.txt

# 6. View file content
cat hello.txt

# 7. Go back one directory
cd ..

# 8. Display tree structure
tree

# 9. Check system information
uname -a

# 10. Display installed Ubuntu version
lsb_release -a
